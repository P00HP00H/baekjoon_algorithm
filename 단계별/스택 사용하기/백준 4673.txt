int number(int n){
a = n + n/(10*digit(n)) + n%(10*digit(n))
if(a>=10000){
return a;
}
else{
number(a);
}
}



void digit(int b)
{
	int c;
	for (c = 0; b != 0; c++) {
		b /= 10;
	}
	printf("%d", c);
	printf("\n");
}
